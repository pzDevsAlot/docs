title: "Commands Guide"

<Badge>Admin Commands</Badge>

<Warning>
  Requires admin permissions as configured in your framework or settings.lua.
</Warning>

<Card title="/pz_wipeall — Force wipe everything" icon="trash">
  Force wipe everything (cars, peds, objects) and reset the timer.
</Card>

<Card title="/pz_wipecars — Wipe Vehicles" icon="car">
  Trigger the Vehicle cleanup module. Resets auto‑timer.
</Card>

<Card title="/pz_wipepeds — Wipe Peds" icon="person">
  Trigger the Ped cleanup module.
</Card>

<Card title="/pz_wipeobjs — Wipe Objects" icon="cubes">
  Trigger the Object cleanup module.
</Card>

<Card title="/nextwipe — Check Timer" icon="clock">
  Shows the time remaining until the next automatic wipe.
</Card>
